    Get a list of hunts from the database, limit 10

      getHunts: function(req, res, next) {
    Hunts.find()
         .limit(10)
         .exec(function(err, results) { 
            if (err) {
              console.log('Error fetching from Hunts DB');
              next(err);
            } else {
              res.queryResults = JSON.stringify(results);
              next();
            }
          });
  },
